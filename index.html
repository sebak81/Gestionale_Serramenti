async function saveFaseLavorazioneConDati() { const nuovaFase = document.getElementById('commessaMacroSelect').value; const op1Select = document.getElementById('workerContattoSelect'); const op2Select = document.getElementById('workerAssegnatoSelect'); const annotazioni = document.getElementById('faseAnnotazioniInput').value.trim(); const op1Value = op1Select.value; const op2Value = op2Select.value; const op1Text = op1Select.options[op1Select.selectedIndex]?.text || 'Nessuno'; const op2Text = op2Select.options[op2Select.selectedIndex]?.text || 'Nessuno'; const t = new Date().toLocaleDateString('it-IT', { day: '2-digit', month: '2-digit', hour: '2-digit', minute:'2-digit' }); // 1. Prepariamo la riga di testo da salvare nel Diario Global let rigaLog = `[${t}] - Fase: ${nuovaFase}. `; if(nuovaFase==='PRIMO_CONTATTO') rigaLog += `Contatto da: ${op1Text} | Assegnato: ${op2Text} `; else rigaLog += `Incaricato: ${op1Text} `; rigaLog += `| Note: ${annotazioni || 'Nessuna nota inserita'}`; // Uniamo la nuova riga al diario esistente (se c'è già) let diarioAggiornato = currentCommessa.note_cantiere ? currentCommessa.note_cantiere + '\n' + rigaLog : rigaLog; // CASO A: Stiamo solo aggiornando i dati della stessa fase senza cambiarla if(nuovaFase === currentCommessa.stato_macro) { await db.from('commesse').update({ contatto_gestito_da: op1Value===""?null:op1Value, preventivo_assegnato_a: op2Value===""?null:op2Value, note_cantiere: diarioAggiornato // <--- SALVA NEL DIARIO GLOBAL }).eq('id', currentCommessaId); currentCommessa.contatto_gestito_da = op1Value; currentCommessa.preventivo_assegnato_a = op2Value; currentCommessa.note_cantiere = diarioAggiornato; document.getElementById('faseAnnotazioniInput').value = ""; fetchDiarioTimeline(); // Aggiorna subito la lista a schermo alert("✅ Diario e Operatori Aggiornati!"); return; } // CASO B: C'è un vero e proprio avanzamento di Fase if(!op1Value || !annotazioni) { alert("Incaricato e Annotazioni obbligatori per cambiare fase!"); return; } await db.from('commesse').update({ stato_macro: nuovaFase, contatto_gestito_da: op1Value, preventivo_assegnato_a: op2Value===""?null:op2Value, note_cantiere: diarioAggiornato // <--- SALVA NEL DIARIO GLOBAL }).eq('id', currentCommessaId); currentCommessa.stato_macro = nuovaFase; currentCommessa.contatto_gestito_da = op1Value; currentCommessa.preventivo_assegnato_a = op2Value; currentCommessa.note_cantiere = diarioAggiornato; document.getElementById('faseAnnotazioniInput').value = ""; alert("🎉 Nuova Fase e Diario Salvati!"); openCommessaWorkspace(currentCommessaId); // Rinfresca tutto il pannello }
